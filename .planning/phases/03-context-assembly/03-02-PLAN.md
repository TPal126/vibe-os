    - Body: text input field for URL
      - Input: `w-full px-3 py-2 bg-v-bg border border-v-border rounded text-[12px] font-mono text-v-text placeholder:text-v-dim focus:border-v-accent focus:outline-none`
      - Placeholder: "https://github.com/org/repo.git"
      - `autoFocus`, `disabled={loading}`
      - `onKeyDown`: Enter triggers submit
    - Error display: `text-[10px] text-v-red mt-2` -- shown when error state is set
    - Footer: flex-end with "Cancel" and "Clone & Add" buttons

      - Submit: `px-3 py-1.5 text-[11px] font-sans font-semibold bg-v-accent/20 text-v-accentHi rounded hover:bg-v-accent/30 disabled:opacity-40`
      - Submit disabled when url is empty or loading is true
      - Submit text: "Cloning..." when loading, "Clone & Add" otherwise

    State: `url` (string), `loading` (boolean), `error` (string | null).

    Language color helper (define at top of file):
    ```typescript
    function languageColor(lang: string): string {
      switch (lang) {
        case "Python": return "text-v-green";
        case "TypeScript": return "text-v-accent";
        case "JavaScript": return "text-v-orange";
        case "Rust": return "text-v-red";
        default: return "text-v-dim";
      }
    }
    function languageBg(lang: string): string {
      switch (lang) {
        case "Python": return "bg-v-green/15";
        case "TypeScript": return "bg-v-accent/15";
        case "JavaScript": return "bg-v-orange/15";
        case "Rust": return "bg-v-red/15";
        default: return "bg-v-border/50";
      }
    }
    ```
