    ```typescript
    const { systemPrompt, composedPrompt, setSystemPrompt, recompose } = useAppStore(
      useShallow((s) => ({
        systemPrompt: s.systemPrompt,
        composedPrompt: s.composedPrompt,
        setSystemPrompt: s.setSystemPrompt,
        recompose: s.recompose,
      }))
    );
    ```

    State:
    - `activeSubTab`: "system" | "task" | "skills" | "repo" -- default "system"
    - `copied`: boolean -- briefly true after copy, resets after 2 seconds
    - `localSystemPrompt`: string -- local state for the textarea, synced with store. Use a ref to track debounce timeout:
      ```typescript
      const [localSystemPrompt, setLocalSystemPrompt] = useState(systemPrompt);
      const debounceRef = useRef<ReturnType<typeof setTimeout>>();
      ```

    Sync local state from store when store changes (e.g., on hydration):
    ```typescript
    // Sync from store when it changes externally (hydration, etc.)
    const storePromptRef = useRef(systemPrompt);
    if (storePromptRef.current !== systemPrompt) {
      storePromptRef.current = systemPrompt;
      setLocalSystemPrompt(systemPrompt);
    }
    ```

    System prompt change handler with debounced persistence:
    ```typescript
    const handleSystemPromptChange = useCallback((value: string) => {
      setLocalSystemPrompt(value);
      if (debounceRef.current) clearTimeout(debounceRef.current);
      debounceRef.current = setTimeout(() => {
        setSystemPrompt(value);
      }, 500); // Debounce 500ms before persisting to SQLite
    }, [setSystemPrompt]);
    ```

    Copy handler:
    ```typescript
    const handleCopy = useCallback(async () => {
      if (!composedPrompt) return;
      try {
        await writeText(composedPrompt.full);
        setCopied(true);
        setTimeout(() => setCopied(false), 2000);
      } catch (err) {
        console.error("Failed to copy to clipboard:", err);
      }
    }, [composedPrompt]);
    ```

    Sub-tab definitions:
    ```typescript
    const subTabs = [
      { id: "system", label: "System" },
      { id: "task", label: "Task" },
      { id: "skills", label: "Skills" },
      { id: "repo", label: "Repo" },
    ] as const;
    ```

    Layout:
    ```
    <div className="flex flex-col h-full overflow-hidden">
      <PanelHeader
        title="PROMPT LAYER"
        icon={<Layers size={12} />}
        actions={
          <div className="flex items-center gap-1.5">
            {/* Total token count */}
            {composedPrompt && (
              <span className="text-[10px] font-mono text-v-dim">
                ~{formatTokens(composedPrompt.totalTokens)} tokens
              </span>
            )}
            {/* Copy button */}
            <IconButton
              icon={copied ? <Check size={12} /> : <Copy size={12} />}
              onClick={handleCopy}
              title="Copy Full Prompt"
              active={copied}
            />
          </div>
        }
      />

      {/* Sub-tab strip */}
      <div className="flex border-b border-v-border bg-v-bg shrink-0">
        {subTabs.map((tab) => (
          <button
            key={tab.id}
            onClick={() => setActiveSubTab(tab.id)}
            className={`px-3 py-1 text-[10px] font-sans tracking-wider uppercase transition-all border-b-2 ${
              activeSubTab === tab.id
                ? "font-semibold text-v-textHi border-v-accent"
                : "font-normal text-v-dim border-transparent hover:text-v-text"
            }`}
          >
            {tab.label}
          </button>
        ))}
      </div>

      {/* Content area */}
      <div className="flex-1 overflow-auto p-2">
        {activeSubTab === "system" ? (
          {/* System prompt: EDITABLE textarea */}
          <textarea
            value={localSystemPrompt}
            onChange={(e) => handleSystemPromptChange(e.target.value)}
            placeholder="Enter your system prompt here..."
            className="w-full h-full resize-none bg-v-bg border border-v-border rounded p-3 text-[11px] font-mono text-v-text placeholder:text-v-dim/50 focus:border-v-accent focus:outline-none"
          />
        ) : (
          {/* Task/Skills/Repo: READONLY textareas */}
          <textarea
            value={
              activeSubTab === "task"
                ? composedPrompt?.task ?? ""
                : activeSubTab === "skills"
                  ? composedPrompt?.skills ?? ""
                  : composedPrompt?.repo ?? ""
            }
            readOnly
            className="w-full h-full resize-none bg-v-bg border border-v-border/50 rounded p-3 text-[11px] font-mono text-v-text/80 cursor-default"
          />
        )}
      </div>
    </div>
    ```

    - The content area uses `p-2` padding, and the textareas fill the space with `w-full h-full`
    - The sub-tab strip uses 10px text with wider tracking (matching the design system's dense panel style)

    **Step 2: Update App.tsx with initialization logic**

    The app needs to initialize on launch: create/load session, load repos, discover skills, and trigger initial prompt composition. This runs once on mount.

    Update `src/App.tsx`:
    ```typescript
    import { useEffect } from "react";
    import { TitleBar } from "./components/layout/TitleBar";
    import { MainLayout } from "./components/layout/MainLayout";
    import { StatusBar } from "./components/layout/StatusBar";
    import { useAppStore } from "./stores";

    function App() {
      const loadActiveSession = useAppStore((s) => s.loadActiveSession);
      const createSession = useAppStore((s) => s.createSession);
      const loadRepos = useAppStore((s) => s.loadRepos);
      const discoverSkills = useAppStore((s) => s.discoverSkills);
      const recompose = useAppStore((s) => s.recompose);

      useEffect(() => {
        async function init() {
          // 1. Try to load an existing active session
          await loadActiveSession();
          const session = useAppStore.getState().activeSession;

          // 2. If no active session, create one
          if (!session) {
            await createSession();
          }

          // 3. Load repos from disk
          await loadRepos();

          // 4. Discover skills from global + project dirs
          await discoverSkills();

          // 5. Compose initial prompt
          await recompose();
        }

        init();
      }, []); // eslint-disable-line react-hooks/exhaustive-deps
      // Intentionally empty deps -- init runs once on mount only.
      // The functions are stable references from Zustand and do not change.

      return (
        <div className="h-screen flex flex-col bg-v-bg overflow-hidden">
          <TitleBar />
          <MainLayout />
          <StatusBar />
        </div>
      );
    }

    export default App;
    ```

    IMPORTANT NOTES:
    - Use `useAppStore.getState()` to read the current state after `loadActiveSession()` completes (not the selector, which would be stale in the closure)
    - The empty dependency array is intentional -- init runs exactly once on app mount
    - Individual store selectors are used (not useShallow) because each is a single function reference, which is already stable
    - The init function is async and called inside useEffect -- this is the correct pattern for async initialization
    - If any init step fails, the error is logged by the slice (catch in each slice action) and the app continues with defaults
